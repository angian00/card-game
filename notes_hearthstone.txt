Filter criteria for supported cards:
	collectible == true
	cardClass == "NEUTRAL"
	set == "CORE" || "EXPERT1"
	rarity == "FREE" || "COMMON"
	type == "MINION" || "SPELL" || "ENCHANTMENT" || "WEAPON"
	<only supported mechanics>
